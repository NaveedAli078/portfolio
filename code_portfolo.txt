// Helper method to implement Exponential Backoff fetch for Gemini LLM API
const executeGeminiCall = async (promptText, systemPromptText = "") => {
  const apiKey = ""; // Provided automatically in runtime environment, keep empty as specified
  const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
  
  const payload = {
    contents: [{ parts: [{ text: promptText }] }]
  };
  if (systemPromptText) {
    payload.systemInstruction = { parts: [{ text: systemPromptText }] };
  }

  const wait = (ms) => new Promise((resolve) => setTimeout(resolve, ms));
  const backoffDelays = [1000, 2000, 4000, 8000, 16000];
  let lastError = null;

  for (let attempt = 0; attempt < 5; attempt++) {
    try {
      const response = await fetch(url, {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
        },
        body: JSON.stringify(payload)
      });

      if (!response.ok) {
        throw new Error(`Gemini API Response Code: ${response.status}`);
      }

      const json = await response.json();
      const extractedText = json.candidates?.[0]?.content?.parts?.[0]?.text;
      
      if (extractedText) {
        return extractedText;
      } else {
        throw new Error("Invalid structure returned from Gemini response payload.");
      }
    } catch (err) {
      lastError = err;
      await wait(backoffDelays[attempt]);
    }
  }
  throw lastError || new Error("Failed to reach Gemini after maximum attempts.");
};

// High-Performance Interactive GeoAI Satellite Simulation Component
function InteractiveSatelliteBackground({ darkMode }) {
  const canvasRef = useRef(null);

  useEffect(() => {
    const canvas = canvasRef.current;
    if (!canvas) return;
    const ctx = canvas.getContext('2d');
    let animationFrameId;
    let width = (canvas.width = canvas.offsetWidth);
    let height = (canvas.height = canvas.offsetHeight);

    // Particle nodes representing telemetry/GIS global connectivity (cyan and emerald)
    const particles = [];
    const particleCount = Math.min(35, Math.floor(width / 35));
    const maxLineDist = 130;

    for (let i = 0; i < particleCount; i++) {
      particles.push({
        x: Math.random() * width,
        y: Math.random() * height,
        vx: (Math.random() - 0.5) * 0.4,
        vy: (Math.random() - 0.5) * 0.4,
        radius: Math.random() * 2 + 1,
        pulseSpeed: 0.02 + Math.random() * 0.03,
        pulseVal: Math.random() * Math.PI,
        colorType: Math.random() > 0.5 ? 'cyan' : 'emerald'
      });
    }

    // Satellite setup
    const satellite = {
      angle: 0,
      orbitRadiusX: Math.min(width * 0.42, 350),
      orbitRadiusY: Math.min(height * 0.28, 140),
      centerX: width * 0.72,
      centerY: height * 0.45,
      speed: 0.0035,
      scale: 1,
    };

    // GeoAI rising coordinates/telemetry tickers
    const geoTickers = [];
    const tickerCount = 5;
    for (let i = 0; i < tickerCount; i++) {
      geoTickers.push({
        x: Math.random() * (width * 0.8),
        y: Math.random() * height,
        text: `${(30 + Math.random() * 5).toFixed(4)}° N, ${(71 + Math.random() * 4).toFixed(4)}° E`,
        speed: 0.2 + Math.random() * 0.25,
        opacity: Math.random() * 0.4 + 0.1,
      });
    }

    const resizeHandler = () => {
      if (!canvas) return;
      width = canvas.width = canvas.offsetWidth;
      height = canvas.height = canvas.offsetHeight;
      satellite.centerX = width * 0.72;
      satellite.centerY = height * 0.45;
      satellite.orbitRadiusX = Math.min(width * 0.42, 350);
      satellite.orbitRadiusY = Math.min(height * 0.28, 140);
    };

    window.addEventListener('resize', resizeHandler);

    const drawSimulation = () => {
      ctx.clearRect(0, 0, width, height);

      // 1. Draw Globe Silhouette Representation (Earth center)
      const earthRadius = Math.min(width * 0.14, 110);
      const earthGradient = ctx.createRadialGradient(
        satellite.centerX,
        satellite.centerY,
        0,
        satellite.centerX,
        satellite.centerY,
        earthRadius
      );
      
      const themeColor1 = 'rgba(0, 200, 83, 0.12)'; // Emerald Green
      const themeColor2 = 'rgba(0, 229, 255, 0.22)'; // Neon Cyan
      
      earthGradient.addColorStop(0, themeColor2);
      earthGradient.addColorStop(0.6, themeColor1);
      earthGradient.addColorStop(1, 'rgba(0, 0, 0, 0)');

      ctx.beginPath();
      ctx.arc(satellite.centerX, satellite.centerY, earthRadius * 1.5, 0, Math.PI * 2);
      ctx.fillStyle = earthGradient;
      ctx.fill();

      // Earth Wireframe Lines
      ctx.strokeStyle = darkMode ? 'rgba(0, 229, 255, 0.05)' : 'rgba(30, 136, 229, 0.1)';
      ctx.lineWidth = 1;
      for (let r = 20; r <= earthRadius; r += 20) {
        ctx.beginPath();
        ctx.arc(satellite.centerX, satellite.centerY, r, 0, Math.PI * 2);
        ctx.stroke();
      }

      // 2. Draw Connection Nodes & Spatial Tensors
      particles.forEach((p, idx) => {
        p.x += p.vx;
        p.y += p.vy;

        // Bounce boundaries
        if (p.x < 0 || p.x > width) p.vx *= -1;
        if (p.y < 0 || p.y > height) p.vy *= -1;

        p.pulseVal += p.pulseSpeed;
        const opacity = (Math.sin(p.pulseVal) + 1) / 2 * 0.35 + 0.15;

        ctx.beginPath();
        ctx.arc(p.x, p.y, p.radius, 0, Math.PI * 2);
        ctx.fillStyle = p.colorType === 'cyan' 
          ? `rgba(0, 229, 255, ${opacity})`
          : `rgba(0, 200, 83, ${opacity})`;
        ctx.fill();

        // Draw connections
        for (let j = idx + 1; j < particles.length; j++) {
          const p2 = particles[j];
          const dist = Math.hypot(p.x - p2.x, p.y - p2.y);
          if (dist < maxLineDist) {
            const lineAlpha = (1 - dist / maxLineDist) * 0.12;
            ctx.beginPath();
            ctx.moveTo(p.x, p.y);
            ctx.lineTo(p2.x, p2.y);
            ctx.strokeStyle = darkMode 
              ? `rgba(0, 229, 255, ${lineAlpha})` 
              : `rgba(30, 136, 229, ${lineAlpha * 1.5})`;
            ctx.lineWidth = 0.8;
            ctx.stroke();
          }
        }
      });

      // 3. Draw Orbit Track Line (Elliptical projection)
      ctx.beginPath();
      ctx.ellipse(
        satellite.centerX,
        satellite.centerY,
        satellite.orbitRadiusX,
        satellite.orbitRadiusY,
        Math.PI * 0.08, // Subtle orbital tilt
        0,
        Math.PI * 2
      );
      ctx.strokeStyle = darkMode ? 'rgba(0, 229, 255, 0.08)' : 'rgba(30, 136, 229, 0.1)';
      ctx.setLineDash([4, 6]);
      ctx.stroke();
      ctx.setLineDash([]); // Reset line dash

      // 4. Calculate Satellite Position
      satellite.angle += satellite.speed;
      const angleTilt = Math.PI * 0.08;
      const rawX = Math.cos(satellite.angle) * satellite.orbitRadiusX;
      const rawY = Math.sin(satellite.angle) * satellite.orbitRadiusY;

      // Project tilted 2D coordinates
      const satX = satellite.centerX + rawX * Math.cos(angleTilt) - rawY * Math.sin(angleTilt);
      const satY = satellite.centerY + rawX * Math.sin(angleTilt) + rawY * Math.cos(angleTilt);

      // 5. Draw Active Satellite Scanning Beam / Sensor Swath
      const beamGrad = ctx.createRadialGradient(satX, satY, 5, satX, satY + 120, 180);
      beamGrad.addColorStop(0, 'rgba(0, 200, 83, 0.18)');
      beamGrad.addColorStop(0.5, 'rgba(0, 229, 255, 0.05)');
      beamGrad.addColorStop(1, 'rgba(0, 0, 0, 0)');

      ctx.beginPath();
      ctx.moveTo(satX, satY);
      ctx.lineTo(satX - 80, satY + 220);
      ctx.lineTo(satX + 80, satY + 220);
      ctx.closePath();
      ctx.fillStyle = beamGrad;
      ctx.fill();

      // Laser scanning dynamic target reticle
      ctx.beginPath();
      ctx.arc(satX, satY + 210, 4, 0, Math.PI * 2);
      ctx.strokeStyle = 'rgba(0, 229, 255, 0.4)';
      ctx.stroke();

      // 6. Draw Detailed Stylized Satellite Node
      ctx.fillStyle = '#00C853';
      ctx.beginPath();
      ctx.arc(satX, satY, 5, 0, Math.PI * 2);
      ctx.fill();

      // Satellite Solar Panels (Side rects)
      ctx.fillStyle = '#1E88E5';
      ctx.fillRect(satX - 16, satY - 2, 9, 4);
      ctx.fillRect(satX + 7, satY - 2, 9, 4);

      // Panel wire outlines
      ctx.strokeStyle = '#FFFFFF';
      ctx.lineWidth = 0.5;
      ctx.strokeRect(satX - 16, satY - 2, 9, 4);
      ctx.strokeRect(satX + 7, satY - 2, 9, 4);

      // Central Antenna Core
      ctx.beginPath();
      ctx.moveTo(satX, satY);
      ctx.lineTo(satX, satY + 7);
      ctx.strokeStyle = '#00C853';
      ctx.lineWidth = 1.5;
      ctx.stroke();

      // Satellite Halo Pulse
      ctx.beginPath();
      ctx.arc(satX, satY, 18 + Math.sin(Date.now() * 0.005) * 4, 0, Math.PI * 2);
      ctx.strokeStyle = 'rgba(0, 229, 255, 0.2)';
      ctx.lineWidth = 1;
      ctx.stroke();

      // 7. Render Geospatial Coord Tickers (Rising coordinates)
      ctx.font = '9px monospace';
      ctx.fillStyle = darkMode ? 'rgba(0, 229, 255, 0.25)' : 'rgba(30, 136, 229, 0.35)';
      geoTickers.forEach((t) => {
        t.y -= t.speed;
        if (t.y < -15) {
          t.y = height + 15;
          t.x = Math.random() * (width * 0.8);
        }
        ctx.fillText(t.text, t.x, t.y);
      });

      animationFrameId = requestAnimationFrame(drawSimulation);
    };

    drawSimulation();

    return () => {
      cancelAnimationFrame(animationFrameId);
      window.removeEventListener('resize', resizeHandler);
    };
  }, [darkMode]);

  return <canvas ref={canvasRef} className="absolute inset-0 w-full h-full -z-10 opacity-70 transition-opacity" />;
}
